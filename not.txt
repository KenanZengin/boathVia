
datasource db {
  provider  = "postgresql"
  url  	    = env("DATABASE_URL")
  directUrl = env("DIRECT_URL")
}



generator client {
  provider = "prisma-client-js"
}

model Ships{
  id String @id @default(cuid()) @unique
  city String
  district String
  star Float
  comment Int
  category String
  capacity Int
  hour_price Int
  img_path String 
  full_hours String[]
}

model Services{
  id String @id @default(cuid()) @unique
  kind String
  info String
  img_path_2 String 
}

enum UserRole {
  ADMIN
  USER
}

enum UserLanguage {
  ENG
  TR
}

model User{
  id String @id @default(cuid())
  name String?
  surname String?
  email String? @unique
  password String?
  phone String?
  rights Boolean @default(false)
  role UserRole @default(USER)
  language UserLanguage @default(ENG)
  image String?
  emailVerified DateTime?
  userFavShips UserFavShips?
}

model UserFavShips {
  id String @id @default(cuid())
  userId String @unique
  shipsId String[]

  user User @relation(fields: [userId],references: [id],onDelete: Cascade)

}
